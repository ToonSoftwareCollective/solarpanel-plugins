	function getSolarData(passWord,userName,apiKey,siteid,urlString,useSSL,totalPower){
		if (debugOutput) console.log("*********SolarPanel Start getFP4ALLData")
		var http = new XMLHttpRequest();
		http.open("GET", "http://" + urlString + "/status.xml", true)
		http.onreadystatechange = function() {
			if (http.readyState == XMLHttpRequest.DONE) {
				if (http.status === 200 || http.status === 300  || http.status === 302) {
					try {
						var response = http.responseText
						var n201 = response.indexOf("<gauge_power>") + "<gauge_power>".length
						var n202 = response.indexOf("<", n201)
						currentPower = response.substring(n201, n202);
						
						var today2
						var n205 = response.indexOf("<energy_today>") + "<energy_today>".length
						var n206 = response.indexOf("<", n205)
						today2 = response.substring(n205, n206);
						
						var n208 = response.indexOf("<energy_total>") + "<energy_total>".length
						var n209 = response.indexOf("<", n208)
						totalValue = response.substring(n208, n209);

						parseReturnData(currentPower,totalValue,today2,0,0,0,0,http.status,"succes")
					}
					catch (e){
						currentPower = 0
						parseReturnData(0,totalValue,todayValue,0,0,0,0, http.status,"error")
					}
				} else {
					if (debugOutput) console.log("*********SolarPanel error: " + http.status)
					parseReturnData(0,totalValue,0,0,0,0,0, http.status,"error")
				}
			}
		}
	http.send();
	}
